    // burger modal menu
    function openModal() {
      let w = window.innerWidth - document.documentElement.clientWidth; // window.innerWidth - document.body.offsetWidth;
      document.body.style.paddingRight = w + "px";
      document.getElementById("modal").style.top = "0px";
      document.body.style.overflow = "hidden";
      document.getElementById("modal-shadow").style.display = "block";
    }
    function closeModal() {
      document.getElementById("modal").style.top = "-1400px";
      document.body.style.overflow= "";
      document.getElementById("modal-shadow").style.display = "none";
      document.body.style.paddingRight = 0;
    }
  </script>
